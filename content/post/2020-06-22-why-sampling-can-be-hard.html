n = 1e3
phi_est = 2*pi*runif(n)

theta_est = sapply(U, function(u) acos(1 - 2*u))

x_est = sin(theta_est)*cos(phi_est)
y_est = sin(theta_est)*sin(phi_est)
z_est = cos(theta_est)

theta_est = runif(n, min=0, max=pi)
phi_est = runif(n, min=0, max=2*pi)

y_est_naive = sin(theta_est)*sin(phi_est)
z_est_naive = cos(theta_est)
